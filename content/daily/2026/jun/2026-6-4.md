#### 清理删除购买记录
```bash
kubectl exec -n zheda-agent deploy/stablepay-mysql -- mysql -ustablepay -pstablepay123 -e "
SET FOREIGN_KEY_CHECKS = 0;
USE stablepay_verification_db;
TRUNCATE TABLE purchase_records;
USE stablepay_payment_db;
TRUNCATE TABLE payment_transactions;
TRUNCATE TABLE payment_idempotency_keys;
TRUNCATE TABLE blockchain_callbacks;
TRUNCATE TABLE gas_subsidies;
USE stablepay_query_db;
TRUNCATE TABLE transaction_records;
SET FOREIGN_KEY_CHECKS = 1;
"
```
验证：
```bash
kubectl exec -n zheda-agent deploy/stablepay-mysql -- mysql -ustablepay -pstablepay123 -e "
SELECT 'purchase_records' AS table_name, COUNT(*) FROM stablepay_verification_db.purchase_records
UNION
SELECT 'payment_transactions', COUNT(*) FROM stablepay_payment_db.payment_transactions
UNION
SELECT 'gas_subsidies', COUNT(*) FROM stablepay_payment_db.gas_subsidies;
"
```
